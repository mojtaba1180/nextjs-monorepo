# üîß Setup & Installation

Complete guide for installing and setting up framework package in monorepo projects.

## üì¶ Package Installation

### 1. Adding to Dependencies

```bash
# In your project's package.json
pnpm add @workspace/framework
```

### 2. Installing Dependencies

```bash
pnpm install
```

## ‚öôÔ∏è Environment Variables

### 1. Creating Environment File

```bash
# Create .env.local file in your project root
touch .env.local
```

### 2. Setting Environment Variables

```bash
# .env.local
NEXT_PUBLIC_REST_API_ENDPOINT=https://your-api-url.com

# Optional: API Timeout (in milliseconds)
NEXT_PUBLIC_API_TIMEOUT=70000

# Optional: Enable/disable React Query Devtools
NEXT_PUBLIC_ENABLE_QUERY_DEVTOOLS=true
```

### 3. Different Environment Examples

```bash
# Development
NEXT_PUBLIC_REST_API_ENDPOINT=http://localhost:3001/api

# Staging
NEXT_PUBLIC_REST_API_ENDPOINT=https://staging-api.example.com

# Production
NEXT_PUBLIC_REST_API_ENDPOINT=https://api.example.com
```

## üèóÔ∏è Setup Provider

### 1. In Next.js App Router

```tsx
// app/layout.tsx
import { FrameworkProvider } from "@workspace/framework";

export default function RootLayout({
  children,
}: {
  children: React.ReactNode;
}) {
  return (
    <html lang="en">
      <body>
        <FrameworkProvider>
          {children}
        </FrameworkProvider>
      </body>
    </html>
  );
}
```

### 2. In Next.js Pages Router

```tsx
// pages/_app.tsx
import { FrameworkProvider } from "@workspace/framework";

export default function App({ Component, pageProps }) {
  return (
    <FrameworkProvider>
      <Component {...pageProps} />
    </FrameworkProvider>
  );
}
```

### 3. With Other Providers

```tsx
// app/layout.tsx
import { FrameworkProvider } from "@workspace/framework";
import { ThemeProvider } from "next-themes";

export default function RootLayout({
  children,
}: {
  children: React.ReactNode;
}) {
  return (
    <html lang="en">
      <body>
        <FrameworkProvider>
          <ThemeProvider
            attribute="class"
            defaultTheme="system"
            enableSystem
          >
            {children}
          </ThemeProvider>
        </FrameworkProvider>
      </body>
    </html>
  );
}
```

## üîß Advanced Configuration

### 1. Custom Query Client

```tsx
// app/layout.tsx
import { FrameworkProvider } from "@workspace/framework";
import { QueryClient } from "@tanstack/react-query";

const customQueryClient = new QueryClient({
  defaultOptions: {
    queries: {
      staleTime: 10 * 60 * 1000, // 10 minutes
      gcTime: 15 * 60 * 1000, // 15 minutes
      retry: 2,
    },
    mutations: {
      retry: 1,
    },
  },
});

export default function RootLayout({
  children,
}: {
  children: React.ReactNode;
}) {
  return (
    <FrameworkProvider queryClient={customQueryClient}>
      {children}
    </FrameworkProvider>
  );
}
```

### 2. Disable Devtools in Production

```tsx
// app/layout.tsx
import { FrameworkProvider } from "@workspace/framework";

export default function RootLayout({
  children,
}: {
  children: React.ReactNode;
}) {
  return (
    <FrameworkProvider 
      enableDevtools={process.env.NODE_ENV === "development"}
    >
      {children}
    </FrameworkProvider>
  );
}
```

## üß™ Testing Setup

### 1. Creating Test Component

```tsx
// components/FrameworkTest.tsx
"use client";

import { useProductsQuery } from "@workspace/framework";

export function FrameworkTest() {
  const { data, isLoading, error } = useProductsQuery({
    pageNo: 1,
    rowCount: 1,
  });

  if (isLoading) return <div>Loading...</div>;
  if (error) return <div>Error: {error.message}</div>;

  return (
    <div>
      <h2>Framework Setup Successful!</h2>
      <p>Products loaded: {data?.data?.entries.length || 0}</p>
    </div>
  );
}
```

### 2. Using in Page

```tsx
// app/test/page.tsx
import { FrameworkTest } from "@/components/FrameworkTest";

export default function TestPage() {
  return (
    <div className="container mx-auto p-8">
      <h1 className="text-2xl font-bold mb-4">Framework Test</h1>
      <FrameworkTest />
    </div>
  );
}
```

## üö® Common Issues

### 1. Module Not Found

```bash
# Error: Module not found: Can't resolve '@workspace/framework'
# Solution: Ensure package is installed
pnpm install
```

### 2. Environment Variables

```bash
# Error: API endpoint not found
# Solution: Check .env.local
echo $NEXT_PUBLIC_REST_API_ENDPOINT
```

### 3. Provider Setup

```tsx
// Error: useQuery must be used within a QueryClient
// Solution: Ensure app is wrapped with FrameworkProvider
<FrameworkProvider>
  <YourApp />
</FrameworkProvider>
```
