## Examples

### Basic Usage

```tsx
import { CustomPagination } from "@workspace/custom-ui"

function MyComponent() {
  const [currentPage, setCurrentPage] = useState(1)
  const totalPages = 10

  return (
    <CustomPagination
      totalPages={totalPages}
      currentPage={currentPage}
      onPageChange={setCurrentPage}
    />
  )
}
```

### With Data Loading

```tsx
import { CustomPagination } from "@workspace/custom-ui"

function ProductList() {
  const [currentPage, setCurrentPage] = useState(1)
  const [products, setProducts] = useState([])
  const [totalPages, setTotalPages] = useState(1)

  const handlePageChange = async (page: number) => {
    setCurrentPage(page)
    // Load data for the new page
    const data = await fetchProducts(page)
    setProducts(data.products)
    setTotalPages(data.totalPages)
  }

  return (
    <div>
      {/* Product list */}
      <div className="grid grid-cols-3 gap-4">
        {products.map(product => (
          <ProductCard key={product.id} product={product} />
        ))}
      </div>
      
      {/* Pagination */}
      <CustomPagination
        totalPages={totalPages}
        currentPage={currentPage}
        onPageChange={handlePageChange}
      />
    </div>
  )
}
```

### With URL State

```tsx
import { CustomPagination } from "@workspace/custom-ui"
import { useRouter, useSearchParams } from "next/navigation"

function SearchResults() {
  const router = useRouter()
  const searchParams = useSearchParams()
  const currentPage = parseInt(searchParams.get('page') || '1')
  const totalPages = 25

  const handlePageChange = (page: number) => {
    const params = new URLSearchParams(searchParams)
    params.set('page', page.toString())
    router.push(`?${params.toString()}`)
  }

  return (
    <CustomPagination
      totalPages={totalPages}
      currentPage={currentPage}
      onPageChange={handlePageChange}
    />
  )
}
```

## API Reference

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `totalPages` | `number` | - | Total number of pages |
| `currentPage` | `number` | - | Current active page (1-based) |
| `onPageChange` | `(page: number) => void` | - | Callback function called when page changes |

## Styling

The component uses Tailwind CSS classes and can be customized by modifying the component styles. The main classes used are:

- `Pagination`: Main container with full width
- `PaginationContent`: Content wrapper with full width
- `PaginationItem`: Individual page item container
- `PaginationLink`: Clickable page number
- `PaginationPrevious/Next`: Navigation buttons
- `PaginationEllipsis`: Ellipsis indicator

## Best Practices

1. **Always provide totalPages**: Make sure to calculate the total number of pages correctly
2. **Handle loading states**: Show loading indicators while fetching new page data
3. **URL synchronization**: Consider syncing pagination state with URL parameters
4. **Accessibility**: The component includes proper ARIA attributes for screen readers
5. **Mobile optimization**: Test on mobile devices to ensure good usability
